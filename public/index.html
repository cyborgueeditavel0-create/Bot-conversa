// ===================== QR CODE FUNCIONAL (WhatsApp REAL) ===================== function generateQR() { const qrBox = document.getElementById('qrcode'); qrBox.innerHTML = ''; // LINK DIRETO DO WHATSAPP WEB - FUNCIONAL! const waLink = 'https://web.whatsapp.com/'; try { new QRCode(qrBox, { text: waLink, width: 168, height: 168, colorDark: "#000000", colorLight: "#ffffff", correctLevel: QRCode.CorrectLevel.M }); notify('✅ QR Code gerado! Escaneie com o WhatsApp', 'green'); // Cria link clicável const linkDiv = document.createElement('div'); linkDiv.style.marginTop = '12px'; linkDiv.innerHTML = `🔗 Clique aqui para abrir WhatsApp Web`; qrBox.parentElement.appendChild(linkDiv); } catch(e) { qrBox.innerHTML = `
📱
${waLink}
Clique no link para conectar
`; notify('✅ Clique no link para conectar o WhatsApp', 'green'); } }
